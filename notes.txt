================================= 1. ==============================================
// Basic
# 1. नया folder बनाओ और उसमें जाओ
mkdir myproject
cd myproject

# 2. Git repo initialize करो
git init

# 3. एक नई file बनाओ (example)
echo "Hello Git" > hello.txt

# 4. Repo की स्थिति check करो
git status

# 5. File को staging area में डालो
git add hello.txt
# (या सब फाइलें add करनी हों तो: git add .)

# 6. First commit बनाओ
git commit -m "First commit: added hello.txt"

# 7. GitHub पर नई repo बनाओ (manual step GitHub website पर)

# 8. Remote repo से connect करो
git remote add origin https://github.com/username/myproject.git

# 9. Main branch set करो
git branch -M main

# 10. Code GitHub पर push करो
git push -u origin main



================================= 2. Branch ==============================================

# 1. Branch list देखना
git branch

# 2. नई branch बनाना
git branch feature1

# 3. उस branch पर switch करना
git checkout feature1

# (Shortcut: branch बनाना + switch करना एक साथ)
git checkout -b feature2

# 4. फाइल add करके commit करना (branch में काम करो)
echo "This is feature work" > feature.txt
git add feature.txt
git commit -m "Added feature.txt in feature branch"

# 5. Branch को GitHub पर push करना
git push -u origin feature1

# 6. वापस main branch पर आना
git checkout main

# 7. feature1 branch को main में merge करना
git merge feature1

# 7.1 Merge result को GitHub पर push करना (important)
git push origin main

# 8. Branch delete करना (local)
git branch -d feature1

# 9. Branch delete करना (remote/GitHub)
git push origin --delete feature1

🔹 3. अगर दोनों जगह से हटाना है
तो पहले local delete करो और फिर remote delete:
git branch -d branch_name
git push origin --delete branch_name
